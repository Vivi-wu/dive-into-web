### 异步写入文件

1. 覆盖式

```js
var fs = require('fs');
var result = { 'id': 1 };

fs.writeFile('./result.json', JSON.stringify(result), (err) => {
  if (err) {
    // 报错：
    // 意外错误
    // 文件权限问题
    // 文件夹找不到(不会自动创建文件夹)
    console.log(err);
  } else {
    console.log('success');
  }
});
```
